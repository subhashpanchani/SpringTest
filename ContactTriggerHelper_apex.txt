public class ContactTriggerHelper{

    /*
    * Parameter : contact list
    * This method update account records.
    */
    public static void updateAccounts(List<Contact> contacts){
        Set<Id> acctIds = new Set<Id>();
        for (Contact c : contacts) {
         //Check contact related accountId is not null  
            if (c.AccountId != null) {
                acctIds.add(c.AccountId);
            }
        }
        
        List<Account> acctsToRollup = new List<Account>();
        //Count no of account related to contact and update account fields
        for (AggregateResult ar : [SELECT AccountId AcctId, Count(id) ContactCount 
                                   FROM Contact 
                                   WHERE AccountId in: acctIds 
                                   GROUP BY AccountId]){
            Account act = new Account();
            act.Id = (Id) ar.get('AcctId'); 
            act.Number_of_Contacts__c = (Integer)ar.get('ContactCount');
            acctsToRollup.add(act);
        }
        //Update contact related account
        update acctsToRollup;
    
    }
}
